The is_valid_token used here is slightly different from the one used in class. It only returns true if there are at least 3 consecutive letters in the token.

In [6]:
def is_valid_token(token):
    # keep hashtag, user mentions
    if ((token.startswith('#') or token.startswith('@')) and len(token) > 2):
        return False
    # if a token contains at least two consective English letters, we assume it's a word
    if token.startswith('u+'):
        return False
    if re.match(r'.*[a-z]{3,}.*', token):
        return True
    return False

for char, tweets in chartweetdict.items():
    
    cleanedtweets = []

    for tweet in tweets:
        cleaned_tweet = []
        for token in tweet:
            if token not in words_to_remove:
                cleaned_tweet.append(token)
        cleanedtweets.append(cleaned_tweet)

    chartweetdict[char] = cleanedtweets
